    def write_to(path, str, &block)
      file = "#{File.basename path}.#{ProxyFS.rand32}"
      
      wrapper = lambda do
        open(File.join(File.dirname(__FILE__), "../tmp/log", file), "w") { |stream| stream.write str }

        block.call
      end

      replicate(wrapper, :command => "write_to", :path => path, :file => file)
    end
